// Example Fetch API request to get and display JSON in the iframe
        fetch('https://jsonplaceholder.typicode.com/posts')
            .then(response => response.json()) // Parse the JSON in the response
            .then(data => {
                // Convert the JSON object to a string and format it for display
                var formattedData = JSON.stringify(data, null, 2);
                // Display the formatted data in the iframe
                document.getElementById('output').contentWindow.document.body.innerHTML = "<pre>" + formattedData + "</pre>";
            })
            .catch(error => {
                console.error('There was a problem with the fetch operation:', error);
            });
